import random
from diffusers import StableDiffusionXLPipeline
import torch
import pandas as pd


numPerros= int(input("Cuantos perros quieres que genere \n"))
numGatos= int(input("Cuantos gatos quieres que genere \n"))

acciones= input("Â¿Que acciones haran? \n").split(" ")
numAcciones= len(acciones)-1
prompts= []
for i in range(numPerros):
    prompts.append([f"A dog {acciones[random.randint(0, numAcciones)]}ing", "Dog"])
for i in range(numGatos):
    prompts.append([f"A cat {acciones[random.randint(0, numAcciones)]}ing", "Cat"])


pipe = StableDiffusionXLPipeline.from_pretrained("segmind/SSD-1B", torch_dtype=torch.float16, use_safetensors=True, variant="fp16")
pipe.to("cuda")

df= pd.DataFrame(columns = ["Url", "Type"])
num=1
for i in prompts:
    image= pipe(prompt=i[0]).images[0]
    image.save(f"imagen{num}.jpg")
    df.add([f"imagen{num}.jpg", i[1]])
    num+=1
df.to_csv("Animales.csv")
